$DIGIT [0-9]
$NON-ZERO [^0] IN $DIGIT
$SMALLCASE [a-z]
$LETTER [A-Za-z]
$UPPERCASE [^a-z] IN $LETTER

$CONSTANT $UPPERCASE ( $UPPERCASE | $DIGIT ) *
$FLOAT ( ( $NON-ZERO ( $DIGIT ) * ) | $DIGIT ) \. ( $DIGIT ) * $NON-ZERO
$ASSIGN =
$SEMICOLON ;

$MUL \*
